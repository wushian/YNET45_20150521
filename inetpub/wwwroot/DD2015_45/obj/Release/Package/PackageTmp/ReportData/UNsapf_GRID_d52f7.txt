protected bool UpdateDataAll(string st_ActKey, ref string st_errmsg)
{
  bool bl_updateok = false;
  bool bl_Mod = false;
  //
  string st_ctrl = "", st_ctrlname = "";
  string st_sapf_gkey="", st_sapf_mkey="",;
  DataRow mod_row;
  DataRow[] sel_rows;
  //
  st_ctrl = st_ContentPlaceHolder +"gr_GridView_sapf$ctl"; 
  CmdQueryS.CommandText = " and a.gkey='"+hh_qkey.Value+"'";
  DataTable tb_sapf = new DataTable();
  DAC_sapf sapfDao = new DAC_sapf(conn);
  string st_addselect="";
  string st_addjoin="";
  string st_addunion="";
  DbDataAdapter da_ADP =sapfDao.GetDataAdapter(ApVer, "UNsapf", "sapf", st_addselect, false, st_addjoin, CmdQueryS, st_addunion, "");
  da_ADP.Fill(tb_sapf);
  //
  OleDbTransaction thistran;
  conn.Open();
  thistran = conn.BeginTransaction(IsolationLevel.ReadCommitted);
  da_ADP.UpdateCommand.Transaction = thistran;
  da_ADP.DeleteCommand.Transaction = thistran;
  da_ADP.InsertCommand.Transaction = thistran;
  try
  {
    for (int in_g = 0; in_g <= gr_GridView_sapf.Rows.Count + 4; in_g++)
    {
      //gkey
      st_ctrlname = st_ctrl +DAC.GetGridViewRowId(in_g)+"$tx_sapf_gkey02";
      if (FindControl(st_ctrlname) != null)
      {
        //gkey
        st_ctrlname = st_ctrl +DAC.GetGridViewRowId(in_g)+"$tx_sapf_gkey02";
        st_sapf_gkey = ((HtmlInputHidden)FindControl(st_ctrlname)).Value.Trim();
        //mkey
        st_ctrlname = st_ctrl +DAC.GetGridViewRowId(in_g)+"$tx_sapf_mkey02";
        st_sapf_mkey = ((HtmlInputHidden)FindControl(st_ctrlname)).Value.Trim();
        //
        bl_Mod = true;
      }
      else
      {
        bl_Mod = false;
      }
      //
      if (bl_Mod) 
      {
        sel_rows = tb_sapf.Select("sapf_gkey='"+st_sapf_gkey+"'");
        if (sel_rows.Length == 1) 
        {
          mod_row = sel_rows[0];
          if (
          )
          {
            mod_row.EndEdit();
            st_ActKey=DAC.get_guidkey();  //
          }
        }  //sel_rows.Length == 1
      }  //bl_Mod
    }  //for
    da_ADP.Update(tb_sapf);
    thistran.Commit();
    bl_updateok=true;
  }  //try

  catch (Exception e)
  {
    thistran.Rollback();
    bl_updateok = false;
    st_errmsg += e.Message;
  }
  finally
  {
    thistran.Dispose();
    sapfDao.Dispose();
    tb_sapf.Dispose();
    da_ADP.Dispose();
  }
  return bl_updateok;
}
