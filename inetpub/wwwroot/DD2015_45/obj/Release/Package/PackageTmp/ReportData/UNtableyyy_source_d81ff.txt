    protected void Page_Load(object sender, EventArgs e)

      {

        {

          Session["fmtableyyy_CmdQueryS"] = CmdQueryS;
          Set_Control();

          Session["fmtableyyy_gr_GridView_tableyyy_SelectedIndex"] = gr_GridView_tableyyy.SelectedIndex;

          {
            BindNew(false);
          }

    private void BindNew(bool bl_showdata)
    {
      string SelDataKey = "";
      DataRow[] SelDataRow;
      DataRow CurRow;
      //
      try
      {
        CmdQueryS = (OleDbCommand)Session["fmtableyyy_CmdQueryS"];
      }
      catch
      {
        CmdQueryS.CommandText = "";
      }
      //
      DataTable tb_tableyyy = new DataTable();
      DAC_tableyyy tableyyyDao = new DAC_tableyyy(conn);
      OleDbDataAdapter ad_DataDataAdapter;
      string st_addselect = "";
      string st_addjoin = "";
      string st_addunion = "";
      ad_DataDataAdapter = tableyyyDao.GetDataAdapter(ApVer, "UNtableyyy", "tableyyy", st_addselect, false, st_addjoin, CmdQueryS, st_addunion, "order by xxx ");
      ad_DataDataAdapter.Fill(tb_tableyyy);
      //
      if (tb_tableyyy.Rows.Count > 0)
      {
        bt_05.OnClientClick ="return btnDEL_c()";
        bt_04.OnClientClick = "return btnMOD_c()";
      }
      else
      {
        bt_05.OnClientClick = "return btnDEL0_c()";
        bt_04.OnClientClick = "return btnMOD0_c()";
      }
      gr_GridView_tableyyy.DataSource = tb_tableyyy;
      //fmsn101_GV1_SelectedIndex
      //fmsn101_GV1_PageIndex
      gr_GridView_tableyyy = clsGV.BindGridView(gr_GridView_tableyyy, tb_tableyyy, hh_GridCtrl, ref hh_GridGkey,"fmtableyyy_gr_GridView_tableyyy");
      gr_GridView_tableyyy.DataBind();
      SelDataKey = "tableyyy_gkey='"+hh_GridGkey.Value+"'";
      SelDataRow = tb_tableyyy.Select(SelDataKey);
      //
      if (bl_showdata)
      {
        if (SelDataRow.Length == 1)
        {
          CurRow = SelDataRow[0];
          Session["fmtableyyy_gr_GridView_tableyyy_GridGkey"]=hh_GridGkey.Value;
          BindText(CurRow);
        }
        else
        {
          hh_GridCtrl.Value = "init";
          ClearText();
        }
      }
      tb_tableyyy.Dispose();
      tableyyyDao.Dispose();
    }

    protected void actDEL()
    {
      if (sFN.checkAccessFunc(UserGkey, st_object_func, 5, UserLoginGkey, ref li_AccMsg))
      {
        bool bl_delok = false;
        Set_Control();
        hh_ActKey.Value = DAC.get_guidkey();
        //
        DAC_tableyyy tableyyyDao = new DAC_tableyyy(conn);
        string st_addselect = "";
        string st_addjoin = "";
        string st_addunion = "";
        string st_SelDataKey = "tableyyy_gkey='" + hh_GridGkey.Value + "' and tableyyy_mkey='"+hh_mkey.Value+"' ";
        DataTable tb_tableyyy = new DataTable();
        DbDataAdapter da_ADP = tableyyyDao.GetDataAdapter(ApVer, "UNtableyyy", "tableyyy", st_addselect, false, st_addjoin, CmdQueryS, st_addunion, "");
        da_ADP.Fill(tb_tableyyy);
        DataRow[] DelRow = tb_tableyyy.Select(st_SelDataKey);
        if (DelRow.Length == 1)
        {
          conn.Open();
          OleDbTransaction thistran = conn.BeginTransaction(IsolationLevel.ReadCommitted);
          da_ADP.DeleteCommand.Transaction = thistran;
          try
          {
            tableyyyDao.Insertbalog(conn, thistran,"tableyyy",hh_ActKey.Value, hh_GridGkey.Value);
            tableyyyDao.Insertbtlog(conn, thistran,"tableyyy", DAC.GetStringValue(DelRow[0]["tableyyy_gkey"]), "D", UserName, DAC.GetStringValue(DelRow[0]["tableyyy_gkey"]));
            DelRow[0].Delete();
            da_ADP.Update(tb_tableyyy);
            thistran.Commit();
            bl_delok = true;
          }
          catch (Exception e)
          {
            thistran.Rollback();
            bl_delok = false;
            lb_ErrorMessage.Visible = true;
            lb_ErrorMessage.Text = e.Message;
          }
          finally
          {
            thistran.Dispose();
            tableyyyDao.Dispose();
            tb_tableyyy.Dispose();
            da_ADP.Dispose();
            conn.Close();
          }
        }
        else
        {
          bl_delok = false;
          lb_ErrorMessage.Visible = true;
          lb_ErrorMessage.Text = StringTable.GetString("資料已變更,請重新選取!");
        }
        tb_tableyyy.Clear();
        //
        if (bl_delok)
        {
          gr_GridView_tableyyy = clsGV.SetGridCursor("del", gr_GridView_tableyyy, -2);
          BindNew(true);
          SetSerMod();
        }
        //bl_delok
      }
    }

    protected void actSAV()
    {

      string st_tempgkey = "";
      bool bl_insok = false, bl_updateok = false;
      //
      Set_Control();
      if (ServerEditCheck(ref st_ckerrmsg))
      {
        DAC_tableyyy tableyyyDao = new DAC_tableyyy(conn);
        if (hh_GridCtrl.Value.ToLower() == "modall")
        {
          if (UpdateDataAll(hh_ActKey.Value, ref st_dberrmsg))

            hh_GridCtrl.Value = "rekey";
            BindNew(true);
            hh_GridCtrl.Value = "ser";

          }

        else
        {
          string st_addselect = "";
          string st_addjoin = "";
          string st_addunion = "";
          string st_SelDataKey = "tableyyy_gkey='"+hh_GridGkey.Value+"'";
          if (hh_GridCtrl.Value.ToLower() == "ins")
          {
            //自動編號
            //DateTime dt_idat=Convert.ToDateTime(tx_riza_RIDAT.Text);
            //st_ren_yymmtext =sFN.strzeroi(dt_idat.Year,4)+sFN.strzeroi(dt_idat.Month,2);
            //st_ren_cls=st_ren_yymmtext;
            //tx_riza_RIREN.Text = rizaDao.GetRenW(conn, st_dd_apx, st_ren_cls, st_ren_cos, st_ren_head, st_ren_yymmtext, in_ren_len, false);
            //conn.Close();
            
            //檢查重複
            if (tableyyyDao.IsExists("tableyyy", "_Primary_key_", Primary_key.Text, ""))
            {
              bl_insok = false;
              st_dberrmsg = StringTable.GetString(Primary_key.Text + ",已存在.");
              
             // tableyyyDao.UpDateRenW(st_dd_apx, st_ren_cls, st_ren_cos, tx_riza_RIREN.Text);
             // st_dberrmsg = StringTable.GetString(tx_riza_RIREN.Text + ",已重新取號.");
             // //tx_riza_RIREN.Text = tableyyyDao.GetRenW(conn, st_dd_apx, st_ren_cls, st_ren_cos, st_ren_head, st_ren_yymmtext, in_ren_len, false);             // tx_riza_RIREN.Text ="";
             // conn.Close();
             
            }
            else
            {
              DataTable tb_tableyyy_ins = new DataTable();
              DbDataAdapter da_ADP_ins = tableyyyDao.GetDataAdapter(ApVer, "UNtableyyy", "tableyyy", st_addselect, false, st_addjoin, CmdQueryS_tableyyy, st_addunion, "");
              da_ADP_ins.Fill(tb_tableyyy_ins);
              conn.Open();
              OleDbTransaction thistran = conn.BeginTransaction(IsolationLevel.ReadCommitted);
              da_ADP_ins.InsertCommand.Transaction = thistran;
              try
              {
                DataRow ins_row = tb_tableyyy_ins.NewRow();
                st_tempgkey = DAC.get_guidkey();
                ins_row["tableyyy_gkey"] = st_tempgkey;       // 
                ins_row["tableyyy_mkey"] = DAC.get_guidkey(); //
                ins_row["tableyyy_YT01"] =tx_tableyyy_YT01.Text.Trim();       // YT01NAME
                
                ins_row["tableyyy_trusr"] = UserGkey;  //
                tb_tableyyy_ins.Rows.Add(ins_row);
                //
                //
                da_ADP_ins.Update(tb_tableyyy_ins);
                // tableyyyDao.UpDateRenW(conn, thistran, st_dd_apx, st_ren_cls, st_ren_cos, tx_riza_RIREN.Text.Trim());
                tableyyyDao.Insertbalog(conn, thistran, "tableyyy", hh_ActKey.Value, hh_GridGkey.Value);
                tableyyyDao.Insertbtlog(conn, thistran, "tableyyy", DAC.GetStringValue(ins_row["tableyyy_gkey"]), "I", UserName, DAC.GetStringValue(ins_row["tableyyy_gkey"]));
                thistran.Commit();
                bl_insok = true;
              }
              catch (Exception e)
              {
                thistran.Rollback();
                bl_insok = false;
                st_dberrmsg = e.Message;
              }
              finally
              {
                thistran.Dispose();
                tableyyyDao.Dispose();
                tb_tableyyy_ins.Dispose();
                da_ADP_ins.Dispose();
                conn.Close();
              }
            }
            if (bl_insok)
            {
              hh_GridGkey.Value = st_tempgkey;
              hh_GridCtrl.Value = "rekey";
              BindNew(true);
              hh_GridCtrl.Value = "ser";
              SetSerMod();
            }
            else
            {
              lb_ErrorMessage.Text = st_dberrmsg;
              lb_ErrorMessage.Visible = true;
            } //bl_insok
          }  //ins
          else if (hh_GridCtrl.Value.ToLower() == "mod")
          {
            if (tableyyyDao.IsExists("tableyyy", "_Primary_key_", Primary_key.Text, "gkey<>'"+hh_GridGkey.Value+"'"))

              st_dberrmsg = StringTable.GetString(Primary_key.Text+",已存在.");

              {
                bl_updateok = false;
                st_dberrmsg = StringTable.GetString("資料已變更,請重新選取!");
              }
              else
              {
                conn.Open();
                OleDbTransaction thistran = conn.BeginTransaction(IsolationLevel.ReadCommitted);
                da_ADP_mod.UpdateCommand.Transaction = thistran;
                try
                {
                  mod_row = mod_rows[0];
                  mod_row.BeginEdit();
                  
                  mod_row["tableyyy_YT01"] =tx_tableyyy_YT01.Text.Trim();       // YT01NAME
                  
                  mod_row["tableyyy_mkey"] = DAC.get_guidkey();        //
                  mod_row["tableyyy_trusr"] =UserGkey;  //
                  mod_row.EndEdit();
                  da_ADP_mod.Update(tb_tableyyy_mod);
                  tableyyyDao.Insertbalog(conn, thistran, "tableyyy", hh_ActKey.Value, hh_GridGkey.Value);
                  tableyyyDao.Insertbtlog(conn, thistran, "tableyyy", DAC.GetStringValue(mod_row["tableyyy_gkey"]), "M", UserName, DAC.GetStringValue(mod_row["tableyyy_gkey"]));
                  thistran.Commit();
                  bl_updateok = true;
                }
                catch (Exception e)
                {
                  thistran.Rollback();
                  bl_updateok = false;
                  st_dberrmsg = e.Message;
                }
                finally
                {
                  thistran.Dispose();
                  tableyyyDao.Dispose();
                  tb_tableyyy_mod.Dispose();
                  da_ADP_mod.Dispose();
                  conn.Close();
                }
              } //mod_rows.Length=1

            {
              //hh_GridCtrl.Value = "rekey";
              BindNew(true);
              hh_GridCtrl.Value = "ser";
              SetSerMod();
            }
            else
            {
              lb_ErrorMessage.Text = st_dberrmsg;
              lb_ErrorMessage.Visible = true;
            } //bl_updateok
          }   //mod
        }  //ins & mod
        tableyyyDao.Dispose();
      }
      else
      {
        lb_ErrorMessage.Visible = true;
        lb_ErrorMessage.Text = st_ckerrmsg;
      }
    }
